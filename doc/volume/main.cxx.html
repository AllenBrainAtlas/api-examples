This is the standard main function of a Qt application. It creates a generic application and the custom UI (MainWindow), then starts the app.

 
int main(int argc, char** argv)
{
  QApplication app(argc, argv);
  
  MainWindow win;
  win.show();
  
  return app.exec();
}
